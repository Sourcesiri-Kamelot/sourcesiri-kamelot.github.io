Implementation

Simple integration in just a few lines:

import createWalletConnect from './src/components/WalletConnect/WalletConnect.js';

const wallet = createWalletConnect('#container', {
  showStatus: true,
  showBalance: true,
  notifications: true
});

wallet.on('connected', (data) => {
  console.log('Connected:', data.account);
});
