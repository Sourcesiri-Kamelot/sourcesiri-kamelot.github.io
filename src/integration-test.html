Integration Code

Example of how to integrate the wallet components:

// Import the wallet component
import createWalletConnect from './src/components/WalletConnect/WalletConnect.js';

// Create wallet instances with different configurations
const mainWallet = createWalletConnect('#wallet-main', {
    showStatus: true,
    showBalance: true,
    notifications: true
});

const headerWallet = createWalletConnect('#wallet-header', {
    showStatus: false,
    showBalance: false,
    notifications: false
});

// Listen to events
mainWallet.on('connected', (data) => {
    console.log('Main wallet connected:', data);
});
